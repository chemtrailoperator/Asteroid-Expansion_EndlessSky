# These jobs are expansions for the vanilla ones already in game. For now, only including high-demand minerals and metals

mission "Mineral Shortages (Osmium)"
	name "Harvested osmium to <planet>"
	job
	repeat
	deadline 7
	description "Factories on <planet> are suffering from a mineral shortage. Harvest <amount> tons of <mineral> and return to <destination> by <date>. Payment is <payment>."
	substitutions
		"<mineral>" "osmium"
		"<amount>" "30"
	to offer
		random < 5
		"outfit (all): Osmium" < 5
	to accept
		"cargo space" >= 30
	source
		government "Republic" "Free Worlds" "Syndicate" "Neutral" "Independent"
		attributes "factory"
	on visit
		dialog phrase "generic mineral on visit"
	on complete
		payment 379500
		outfit "Osmium" -30
		dialog phrase "generic mineral delivery payment"
